\section{Setup}

This note records the replacement for the Paley--Zygmund step in the proof of
Theorem~0.4 from the phase retrieval notes.  We use the notation of
\texttt{RealSPR.tex}.  Thus \(\eta,\xi_2,\xi_3,\dots\) are independent
real-valued random variables, \(\E\eta=0\), \(\norm{\eta}_{L^2}=1\), and for a
fixed \(\delta>0\),
\[
        \E\xi_i=0,\qquad \norm{\xi_i}_{L^2}=1,\qquad
        \delta\le \norm{\xi_i}_{L^1}\le 1-\delta
        \qquad (i\ge2).
\]
For coefficient vectors \(a,b\in\ell^2(\N)\), write
\[
X_a=a_1\eta+\sum_{i\ge2}a_i\xi_i,\qquad
X_b=b_1\eta+\sum_{i\ge2}b_i\xi_i.
\]
For finite-dimensional statements we use the analogous notation
\[
Z=(\eta,\xi_2,\ldots,\xi_n)\in\R^n,\qquad
\langle Z,a\rangle=a_1\eta+\sum_{i=2}^n a_i\xi_i .
\]
Finally,
\[
        \distpm{u}{v}:=\min\{\norm{u-v}_2,\norm{u+v}_2\}.
\]

\section{VC Background}

The following material is the only VC-dimension input needed below.  It is the
same package used in Section 2.2 of the declipping manuscript
\texttt{declipping.pdf}.

\begin{definition}[Shattering and VC dimension]
Let \(\mathcal F\) be a class of Boolean functions on \(\R^n\).  A finite set
\(S\subset\R^n\) is shattered by \(\mathcal F\) if for every map
\(g:S\to\{0,1\}\), there is \(f\in\mathcal F\) such that \(f|_S=g\).  The
VC-dimension \(\vc(\mathcal F)\) is the largest cardinality of a finite set
shattered by \(\mathcal F\).
\end{definition}

\begin{proposition}[VC law of large numbers]\label{prop:vc-lln}
Let \(\mathcal F\) be a class of Boolean functions and let
\(X_1,\ldots,X_m\) be i.i.d. copies of a random vector \(X\).  There is an
absolute constant \(C>0\) such that
\[
\E\sup_{f\in\mathcal F}
\abs{\frac1m\sum_{j=1}^m f(X_j)-\E f(X)}
\le C\sqrt{\frac{\vc(\mathcal F)}{m}}.
\]
Consequently, for every \(t>0\), with probability at least \(1-e^{-t}\),
\[
\sup_{f\in\mathcal F}
\abs{\frac1m\sum_{j=1}^m f(X_j)-\E f(X)}
\le C\sqrt{\frac{\vc(\mathcal F)}{m}}+\sqrt{\frac{t}{2m}}.
\]
\end{proposition}

\begin{proposition}[Intersections and unions]\label{prop:vc-intersection}
Let \(\mathcal F_1,\ldots,\mathcal F_k\) be Boolean classes with
\(\vc(\mathcal F_i)=v_i\).  If \(\mathcal F\) is the class of indicators of
intersections of one set from each \(\mathcal F_i\), or the class of indicators
of unions of one set from each \(\mathcal F_i\), then
\[
\vc(\mathcal F)\le C\Bigl(\sum_{i=1}^k v_i\Bigr)\log(Ck),
\]
where \(C>0\) is absolute.
\end{proposition}

\begin{lemma}[Half-space classes]\label{lem:halfspaces}
For each fixed \(k\), the class of indicators of intersections of at most
\(k\) half-spaces in \(\R^n\) has VC-dimension \(O_k(n)\).  Consequently it
satisfies the uniform law of large numbers in Proposition~\ref{prop:vc-lln}
with error \(O_k(\sqrt{n/m})\).
\end{lemma}

\begin{lemma}[Polynomial threshold classes]\label{lem:poly-vc}
Fix integers \(D,N_x,N_\theta\).  Let \(P(x,\theta)\) range through a family of
real polynomials of degree at most \(D\) in the variables
\((x,\theta)\in\R^{N_x}\times\R^{N_\theta}\), with arbitrary parameter
\(\theta\).  Then the Boolean class
\[
        \bigl\{\mathbf 1_{\{x:P(x,\theta)\ge0\}}:\theta\in\R^{N_\theta}\bigr\}
\]
has VC-dimension \(O_{D}(N_x+N_\theta)\).
In particular, if \(N_x=n\) and \(N_\theta=O(n)\), then the VC-dimension is
\(O_D(n)\).  By Proposition~\ref{prop:vc-intersection}, the same remains true
for intersections of a bounded number of such classes.
\end{lemma}

\begin{remark}
Lemma~\ref{lem:poly-vc} is the standard Warren-type VC bound used in the
phase retrieval notes.  In the proof of Theorem~\ref{thm:finite-pr-vc} below,
the relevant defining inequalities have bounded degree and depend on the
parameters \(u,v\in\R^n\), hence the resulting class has VC-dimension \(O(n)\).
\end{remark}

\section{The New Compactness Input}

The purpose of this section is to formulate the compactness replacement for
the Paley--Zygmund lower bound.  In the original proof of Theorem~0.4, one
uses an \(L_{2+\alpha}\)-hypothesis and Paley--Zygmund to prove
\[
\Prob\left(
 \bigl||\langle Z,u\rangle|-|\langle Z,v\rangle|\bigr|^2
      \ge \theta\,\distpm{u}{v}^2
\right)\ge p.
\]
Under the hypotheses of \texttt{RealSPR.tex}, this probability lower bound is
instead obtained by compactness.

\begin{proposition}[Orthogonal probabilistic modulus gap]\label{prop:orth-prob-gap}
There exist constants \(\theta_0=\theta_0(\delta,\eta)>0\) and
\(p_0=p_0(\delta,\eta)>0\) such that, for every orthogonal pair
\(a,b\in\ell^2(\N)\) with \(\norm{a}_2=\norm{b}_2=1\),
\[
\Prob\left(\bigl||X_a|-|X_b|\bigr|\ge\theta_0\right)\ge p_0 .
\]
\end{proposition}

\begin{proof}[Draft proof]
Suppose this is false.  Then there are orthogonal unit vectors
\(a^{(n)},b^{(n)}\in\ell^2(\N)\) such that, for
\[
X_n=X_{a^{(n)}},\qquad Y_n=X_{b^{(n)}},
\]
one has
\[
\Prob\left(\bigl||X_n|-|Y_n|\bigr|\ge \frac1n\right)\le\frac1n.
\]
Equivalently,
\[
        \bigl||X_n|-|Y_n|\bigr|\to0
        \qquad\text{in probability.}
\]

Apply the reordered profile decomposition from \texttt{RealSPR.tex}.  After
passing to a subsequence and permuting the good coordinates row by row, write
\[
a^{(n)}=a_{\rm head}^{(n)}+a_{\rm tail}^{(n)},\qquad
b^{(n)}=b_{\rm head}^{(n)}+b_{\rm tail}^{(n)},
\]
where the heads converge strongly in \(\ell^2\) to some \(a,b\), while the
tail coefficients are infinitesimal:
\[
\max_{i>m_n}\max\{|a_i^{(n)}|,|b_i^{(n)}|\}\to0 .
\]
The corresponding random vectors
\[
\bigl(X_n^{\rm head},Y_n^{\rm head},X_n^{\rm tail},Y_n^{\rm tail}\bigr)
\]
are tight.  Passing to a further subsequence, they converge in distribution to
\[
        (H_X,H_Y,R_X,R_Y).
\]
The head and tail limits are independent, because they are built from
disjoint independent coordinate families.  The tail law is infinitely
divisible, because it is a weak limit of infinitesimal triangular arrays of
independent random variables.  The convergence in probability of the modulus
gap passes to the limit and gives the exact identity
\[
        |H_X+R_X|=|H_Y+R_Y|\qquad\text{a.s.}
\]

The head-identification argument from \texttt{RealSPR.tex} is now applied at
the limit level.  This is the point of the present replacement: we do not try
to infer the bounded-probe identities directly from convergence in probability
before taking the limit.  After conditioning on a tail value in the support of
\((R_X,R_Y)\), the head law is supported on a translated phase-retrieval cone
\[
        |h_X+r_X|=|h_Y+r_Y|.
\]
Using the bounded linear probes \(S_i\), \(S_\eta\), and the quadratic probes
for the good coordinates exactly as in the head-identification lemma of
\texttt{RealSPR.tex} (equivalently, subtracting the deterministic translation
terms before applying the quadratic probes), one obtains
\[
        a_i^2=b_i^2\quad(i\ge2),\qquad
        a_i a_j=b_i b_j\quad(i\ne j,\ i,j\ge2),
\]
and
\[
        a_1a_j=b_1b_j\qquad(j\ge2).
\]
Therefore either \(a=\tau b\) for some \(\tau\in\{\pm1\}\), or else \(\eta\)
is Rademacher and all good head coordinates vanish.  In the first case,
\[
\norm{a_{\rm head}^{(n)}-\tau b_{\rm head}^{(n)}}_2\to0 .
\]
In the Rademacher-only exceptional case the good head again vanishes, so the
argument below applies with the head removed.

The exact limiting identity forces the infinitely divisible tail law onto one
of the phase-retrieval lines.  Indeed, if the head is nonzero, varying the head
over two support points gives \(R_Y=\tau R_X\) for the same sign \(\tau\).  If
the head is zero, then \(|R_X|=|R_Y|\) a.s.; since the tail law is infinitely
divisible and supported on
\[
\Gamma=\{(s,t): |s|=|t|\}=L_+\cup L_-,
\qquad
L_\tau=\{(s,\tau s):s\in\R\},
\]
the support theorem for infinitely divisible laws on \(L_+\cup L_-\) implies
that \(R_Y=\tau R_X\) a.s. for some sign \(\tau\).

Thus
\[
\sum_{i>m_n}(b_i^{(n)}-\tau a_i^{(n)})\xi_{n,i}
\to0
\qquad\text{in distribution.}
\]
The diffuse tail rigidity lemma from \texttt{RealSPR.tex} gives
\[
\norm{b_{\rm tail}^{(n)}-\tau a_{\rm tail}^{(n)}}_2\to0.
\]
Together with the head convergence this yields
\[
\norm{b^{(n)}-\tau a^{(n)}}_2\to0,
\]
contradicting orthogonality:
\[
\norm{b^{(n)}-\tau a^{(n)}}_2^2
=\norm{a^{(n)}}_2^2+\norm{b^{(n)}}_2^2
-2\tau\ip{a^{(n)}}{b^{(n)}}=2.
\]
This contradiction proves the proposition.
\end{proof}

\begin{remark}[Technical point in the draft]\label{rem:shifted-probes}
The only point that should be checked carefully before inserting the argument
into the paper is the parenthetical phrase ``subtracting the deterministic
translation terms before applying the quadratic probes.''  The clean way to
write it is to formulate a separate shifted head-identification lemma: if an
independent head law satisfies
\[
        |H_X+r_X|=|H_Y+r_Y|
\]
for a tail value \(r\) in the relevant support set, then the same rank-one
identities for the head coefficients follow.  The mixed identities are
immediate from the centered linear probes; the diagonal identity is recovered
by applying the quadratic probe to the centered quadratic observable after the
linear terms in \(H_X,H_Y\) have been removed using the already established
linear identities.
\end{remark}

\section{A Version of Theorem 0.4}

The next statement is the finite-dimensional VC conclusion corresponding to
Theorem~0.4 in the notes, with the Paley--Zygmund step replaced by
Proposition~\ref{prop:orth-prob-gap}.  It is stated in the one-exception
notation of \texttt{RealSPR.tex}.

\begin{theorem}[Finite-dimensional empirical phase retrieval]\label{thm:finite-pr-vc}
Fix \(\delta>0\) and an exceptional variable \(\eta\) as above.  There exist
constants \(C,c,\gamma>0\), depending only on \(\delta\) and on the law of
\(\eta\), with the following property.  Let
\[
        Z=(\eta,\xi_2,\ldots,\xi_n)\in\R^n,
\]
where \(\eta,\xi_2,\ldots,\xi_n\) are independent, \(\E\eta=0\),
\(\norm{\eta}_2=1\), and
\[
        \E\xi_i=0,\qquad \norm{\xi_i}_2=1,\qquad
        \delta\le\norm{\xi_i}_1\le1-\delta\qquad(2\le i\le n).
\]
Let \(Z_1,\ldots,Z_m\) be i.i.d. copies of \(Z\).  If \(m\ge Cn\), then with
probability at least \(1-e^{-cm}\),
\[
\frac1m\sum_{j=1}^m
\left(
  |\langle Z_j,u\rangle|-|\langle Z_j,v\rangle|
\right)^2
\ge
\gamma\,\distpm{u}{v}^{\,2}
\]
for every \(u,v\in\Sph^{n-1}\).
\end{theorem}

\begin{proof}[Draft proof]
By the orthogonal reduction from \texttt{RealSPR.tex}, it is enough to prove
the lower bound after reducing the pair \(u,v\) to the orthogonal case.  Thus
we first consider \(u,v\in\Sph^{n-1}\) with \(u\perp v\).  By
Proposition~\ref{prop:orth-prob-gap}, there are constants
\(\theta_0,p_0>0\), depending only on \(\delta\) and \(\eta\), such that
\[
\Prob\left(
  \bigl||\langle Z,u\rangle|-|\langle Z,v\rangle|\bigr|\ge\theta_0
\right)\ge p_0 .
\]
Equivalently, if
\[
\mathcal E(u,v)
=
\left\{
x\in\R^n:
\bigl||\langle x,u\rangle|-|\langle x,v\rangle|\bigr|^2
\ge \theta_0^2
\right\},
\]
then \(\Prob(Z\in\mathcal E(u,v))\ge p_0\).

The class of indicators \(\mathbf 1_{\mathcal E(u,v)}\), with
\(u,v\in\Sph^{n-1}\), has VC-dimension \(O(n)\).  Indeed, as sets,
\(\mathcal E(u,v)\) can be written as the intersection of two bounded-degree
polynomial threshold conditions:
\[
\left(2|\langle x,u\rangle|\,|\langle x,v\rangle|\right)^2
\le
\left(|\langle x,u\rangle|^2+|\langle x,v\rangle|^2-\theta_0^2\right)^2
\]
and
\[
|\langle x,u\rangle|^2+|\langle x,v\rangle|^2\ge \theta_0^2 .
\]
Lemma~\ref{lem:poly-vc} and Proposition~\ref{prop:vc-intersection} therefore
give \(\vc=O(n)\).

By Proposition~\ref{prop:vc-lln}, with probability at least \(1-e^{-cm}\),
\[
\sup_{u\perp v,\ \norm{u}_2=\norm{v}_2=1}
\left|
\frac1m\sum_{j=1}^m\mathbf 1_{\mathcal E(u,v)}(Z_j)
-
\Prob(Z\in\mathcal E(u,v))
\right|
\le \frac{p_0}{2},
\]
provided \(m\ge Cn\).  Hence, uniformly over orthogonal unit pairs,
\[
\frac1m\sum_{j=1}^m\mathbf 1_{\mathcal E(u,v)}(Z_j)\ge \frac{p_0}{2}.
\]
It follows that
\[
\frac1m\sum_{j=1}^m
\left(
  |\langle Z_j,u\rangle|-|\langle Z_j,v\rangle|
\right)^2
\ge
\theta_0^2\cdot \frac{p_0}{2}
\]
for all orthogonal unit \(u,v\).

Finally, applying the same orthogonal reduction used in \texttt{RealSPR.tex}
to the empirical Hilbert space whose norm is
\[
\norm{w}_{m}^{2}
:=
\frac1m\sum_{j=1}^m |\langle Z_j,w\rangle|^2,
\]
and intersecting with the standard empirical covariance event
\[
\frac12\norm{w}_2^2\le \norm{w}_m^2\le 2\norm{w}_2^2
\qquad(w\in\R^n),
\]
which holds with probability at least \(1-e^{-cm}\) for \(m\ge Cn\), gives the
announced estimate for all \(u,v\in\Sph^{n-1}\), after changing constants.
\end{proof}

\begin{remark}[Relation with the original Theorem 0.4 proof]
In the original proof, the event
\[
\mathcal E(u,v)=
\left\{
\bigl||\langle Z,u\rangle|-|\langle Z,v\rangle|\bigr|^2
\ge \theta\,\distpm{u}{v}^{\,2}
\right\}
\]
was bounded from below by Paley--Zygmund, using an
\(L_{2+\alpha}\)-to-\(L_2\) equivalence.  Here the same VC argument is kept,
but the probability lower bound is supplied by the compactness proposition
instead.  This is the precise place where the new tool replaces
Paley--Zygmund.
\end{remark}
